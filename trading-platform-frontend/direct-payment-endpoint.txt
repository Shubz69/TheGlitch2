/**
 * Direct Stripe checkout endpoint that doesn't require authentication
 * This allows the frontend to redirect users directly to Stripe without auth failures
 */
exports.checkoutDirect = async (req, res) => {
  try {
    // Extract courseId from query params
    const { courseId } = req.query;
    
    if (!courseId) {
      return res.status(400).json({ error: 'Course ID is required' });
    }
    
    // Get course from database
    const course = await Course.findById(courseId);
    
    if (!course) {
      return res.status(404).json({ error: 'Course not found' });
    }
    
    // Create Stripe checkout session
    const session = await stripe.checkout.sessions.create({
      payment_method_types: ['card'],
      line_items: [
        {
          price_data: {
            currency: 'usd',
            product_data: {
              name: course.title,
              description: course.description,
            },
            unit_amount: Math.round(course.price * 100), // Convert to cents
          },
          quantity: 1,
        },
      ],
      mode: 'payment',
      success_url: `${process.env.FRONTEND_URL}/payment-success?courseId=${courseId}&session_id={CHECKOUT_SESSION_ID}`,
      cancel_url: `${process.env.FRONTEND_URL}/courses`,
    });
    
    // Instead of returning JSON, redirect directly to Stripe
    return res.redirect(303, session.url);
  } catch (error) {
    console.error('Error creating checkout session:', error);
    return res.status(500).json({ error: 'Failed to create checkout session' });
  }
};

// BACKEND CODE: Add this to your server/routes/payment.routes.js file

/**
 * @route GET /api/payments/checkout-direct
 * @desc Direct checkout endpoint that redirects to Stripe without authentication
 * @access Public
 */
router.get('/checkout-direct', paymentController.checkoutDirect);
